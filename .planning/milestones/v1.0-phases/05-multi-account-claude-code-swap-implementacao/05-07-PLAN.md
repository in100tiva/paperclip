    ```typescript
    import { Router } from "express";
    import { z } from "zod";
    import type { Db } from "@paperclipai/db";
    import { activityLog } from "@paperclipai/db";
    import { and, desc, eq } from "drizzle-orm";
    import { claudeAccountsService } from "../services/claude-accounts.js";
    import { ACTIVITY_ACTION_CLAUDE_ACCOUNT_ROTATED } from "../services/activity-log.js";
    import { validate } from "../middleware/validate.js";
    import { assertCompanyAccess, getActorInfo, assertCompanyAdmin } from "./authz.js";
    import { badRequest, forbidden, notFound } from "../errors.js";

    const createAccountSchema = z.object({
      label: z.string().trim().min(1).max(100),
      configDirSlug: z.string().trim().regex(/^[a-z0-9][a-z0-9-]{0,62}$/, "Slug deve ser lowercase + dash, 1-63 chars"),
    });

    const patchAccountSchema = z.object({
      status: z.enum(["live", "disabled"]).optional(),
      label: z.string().trim().min(1).max(100).optional(),
    });

    export function claudeAccountsRoutes(db: Db) {
      const router = Router();
      const svc = claudeAccountsService(db);

      // GET list
      router.get("/companies/:companyId/claude-accounts", async (req, res) => {
        await assertCompanyAccess(db, req, req.params.companyId);
        const accounts = await svc.listAccounts(req.params.companyId);
        res.json({ accounts });
      });

      // POST create
      router.post(
        "/companies/:companyId/claude-accounts",
        validate(createAccountSchema),
        async (req, res) => {
          // assertCompanyAdmin: only owner/admin can create accounts (D-29)
          await assertCompanyAdmin(db, req, req.params.companyId);
          const actor = await getActorInfo(req);
          if (!actor.userId) {
            throw forbidden("User authentication required to register Claude account");
          }
          const body = createAccountSchema.parse(req.body);
          // svc não tem create (D-09 lista 6 métodos sem create); inline insert here OR add to service.
          // Decisão: inline para v1; refactor para service se necessário em iteração.
          // Inline implementation:
          const { claudeAccounts } = await import("@paperclipai/db");
          const [created] = await db
            .insert(claudeAccounts)
            .values({
              companyId: req.params.companyId,
              ownerUserId: actor.userId,
              label: body.label,
              configDirSlug: body.configDirSlug,
              status: "live",
            })
            .returning();
          res.status(201).json({ account: created });
        },
      );

      // PATCH (toggle status / rename)
      router.patch(
        "/companies/:companyId/claude-accounts/:accountId",
        validate(patchAccountSchema),
        async (req, res) => {
          await assertCompanyAdmin(db, req, req.params.companyId);
          const body = patchAccountSchema.parse(req.body);
          const { claudeAccounts } = await import("@paperclipai/db");
          const updates: Record<string, unknown> = { updatedAt: new Date() };
          if (body.status !== undefined) updates.status = body.status;
          if (body.label !== undefined) updates.label = body.label;

          const [updated] = await db
            .update(claudeAccounts)
            .set(updates)
            .where(
              and(
                eq(claudeAccounts.id, req.params.accountId),
                eq(claudeAccounts.companyId, req.params.companyId),
              ),
            )
            .returning();
          if (!updated) throw notFound("Claude account not found");
          res.json({ account: updated });
        },
      );

      // GET rotation history
      router.get("/companies/:companyId/claude-accounts/rotation-history", async (req, res) => {
        await assertCompanyAccess(db, req, req.params.companyId);
        const limit = Math.min(Number(req.query.limit ?? 50), 200);
        const rows = await db
          .select()
          .from(activityLog)
          .where(
            and(
              eq(activityLog.companyId, req.params.companyId),
              eq(activityLog.action, ACTIVITY_ACTION_CLAUDE_ACCOUNT_ROTATED),
            ),
          )
          .orderBy(desc(activityLog.createdAt))
          .limit(limit);
        res.json({ entries: rows });
      });

      return router;
    }
    ```

    **Helpers cliente: `ui/src/api/claude-accounts.ts`**

    ```typescript
    import { apiRequest } from "./client";

    export interface ClaudeAccount {
      id: string;
      companyId: string;
      ownerUserId: string;
      label: string;
      configDirSlug: string;
      status: "live" | "exhausted" | "cooldown" | "disabled";
      lastQuotaWindowsJson: Record<string, { exhaustedUntil: string; lastTriggeredAt: string; count: number }>;
      exhaustedUntil: string | null;
      lastUsedAt: string | null;
      createdAt: string;
      updatedAt: string;
    }

    export interface RotationHistoryEntry {
      id: string;
      companyId: string;
      action: "claude_account_rotated";
      entityType: "agent";
      entityId: string;
      details: {
        agentId: string;
        fromAccountId: string;
        toAccountId: string;
        reason: "exhausted" | "manual" | "cooldown_expired";
        errorFamily: string | null;
        retryNotBefore: string | null;
        swapStrategy: "resume" | "fallback_full_context" | null;
      };
      createdAt: string;
    }

    export const claudeAccountsApi = {
      list: (companyId: string) =>
        apiRequest<{ accounts: ClaudeAccount[] }>(`/api/companies/${companyId}/claude-accounts`),
      create: (companyId: string, body: { label: string; configDirSlug: string }) =>
        apiRequest<{ account: ClaudeAccount }>(`/api/companies/${companyId}/claude-accounts`, {
          method: "POST",
          body: JSON.stringify(body),
          headers: { "Content-Type": "application/json" },
        }),
      patch: (companyId: string, accountId: string, body: { status?: ClaudeAccount["status"]; label?: string }) =>
        apiRequest<{ account: ClaudeAccount }>(`/api/companies/${companyId}/claude-accounts/${accountId}`, {
          method: "PATCH",
          body: JSON.stringify(body),
          headers: { "Content-Type": "application/json" },
        }),
      rotationHistory: (companyId: string, limit = 50) =>
        apiRequest<{ entries: RotationHistoryEntry[] }>(
          `/api/companies/${companyId}/claude-accounts/rotation-history?limit=${limit}`,
        ),
    };
    ```

    Estrutura mínima (replicando pattern de CompanyInvites.tsx):

    ```typescript
    import { useState } from "react";
    import { useQuery, useMutation, useQueryClient } from "@tanstack/react-query";
    import { claudeAccountsApi, type ClaudeAccount } from "@/api/claude-accounts";
    import { Button } from "@/components/ui/button";
    import { useBreadcrumbs } from "@/context/BreadcrumbContext";
    import { useCompany } from "@/context/CompanyContext";
    import { useToast } from "@/context/ToastContext";
    import { queryKeys } from "@/lib/queryKeys";

    function StatusBadge({ status }: { status: ClaudeAccount["status"] }) {
      const colorMap = {
        live: "bg-green-500",
        exhausted: "bg-red-500",
        cooldown: "bg-yellow-500",
        disabled: "bg-gray-400",
      };
      return <span className={`inline-block px-2 py-1 rounded text-white text-xs ${colorMap[status]}`}>{status}</span>;
    }

    export function ClaudeAccounts() {
      const { selectedCompanyId } = useCompany();
      const { setBreadcrumbs } = useBreadcrumbs();
      const { pushToast } = useToast();
      const queryClient = useQueryClient();
      const [newLabel, setNewLabel] = useState("");
      const [newSlug, setNewSlug] = useState("");

      // ... breadcrumbs setup ...

      const accountsQuery = useQuery({
        queryKey: ["claudeAccounts", selectedCompanyId],
        queryFn: () => claudeAccountsApi.list(selectedCompanyId!),
        enabled: Boolean(selectedCompanyId),
      });

      const historyQuery = useQuery({
        queryKey: ["claudeAccounts", selectedCompanyId, "history"],
        queryFn: () => claudeAccountsApi.rotationHistory(selectedCompanyId!, 50),
        enabled: Boolean(selectedCompanyId),
      });

      const createMutation = useMutation({
        mutationFn: (body: { label: string; configDirSlug: string }) =>
          claudeAccountsApi.create(selectedCompanyId!, body),
        onSuccess: () => {
          pushToast({ message: "Account registered. Run `claude login` in that dir to activate.", variant: "success" });
          setNewLabel("");
          setNewSlug("");
          queryClient.invalidateQueries({ queryKey: ["claudeAccounts", selectedCompanyId] });
        },
        onError: (err) => {
          pushToast({ message: `Failed to register: ${err instanceof Error ? err.message : "unknown error"}`, variant: "error" });
        },
      });

      const toggleMutation = useMutation({
        mutationFn: (input: { accountId: string; status: ClaudeAccount["status"] }) =>
          claudeAccountsApi.patch(selectedCompanyId!, input.accountId, { status: input.status }),
        onSuccess: () => {
          queryClient.invalidateQueries({ queryKey: ["claudeAccounts", selectedCompanyId] });
        },
      });

      // Render: form de registro, tabela de accounts, lista de history
      return (
        <div className="p-6">
          <h1 className="text-2xl font-semibold mb-4">Claude Accounts</h1>

          <section className="mb-8">
            <h2 className="text-lg font-medium mb-2">Register new account</h2>
            <form
              onSubmit={(e) => {
                e.preventDefault();
                createMutation.mutate({ label: newLabel, configDirSlug: newSlug });
              }}
              className="space-y-2"
            >
              <input
                type="text"
                placeholder="Label (e.g., 'Account A')"
                value={newLabel}
                onChange={(e) => setNewLabel(e.target.value)}
                required
                className="border px-2 py-1 rounded"
              />
              <input
                type="text"
                placeholder="Config dir slug (e.g., 'a')"
                value={newSlug}
                onChange={(e) => setNewSlug(e.target.value)}
                required
                pattern="^[a-z0-9][a-z0-9-]{0,62}$"
                className="border px-2 py-1 rounded"
              />
              <p className="text-xs text-gray-500">
                After registering, run <code>claude login</code> in <code>~/.paperclip/claude-accounts/&lt;slug&gt;/</code>
              </p>
              <Button type="submit" disabled={createMutation.isPending}>
                {createMutation.isPending ? "Registering..." : "Register"}
              </Button>
            </form>
          </section>

          <section className="mb-8">
            <h2 className="text-lg font-medium mb-2">Accounts</h2>
            {accountsQuery.isLoading ? (
              <p>Loading...</p>
            ) : accountsQuery.data?.accounts.length === 0 ? (
              <p className="text-gray-500">No accounts registered yet.</p>
            ) : (
              <table className="w-full">
                <thead>
                  <tr><th>Label</th><th>Slug</th><th>Status</th><th>Last used</th><th>Exhausted until</th><th>Actions</th></tr>
                </thead>
                <tbody>
                  {accountsQuery.data?.accounts.map((acc) => (
                    <tr key={acc.id}>
                      <td>{acc.label}</td>
                      <td><code>{acc.configDirSlug}</code></td>
                      <td><StatusBadge status={acc.status} /></td>
                      <td>{acc.lastUsedAt ? new Date(acc.lastUsedAt).toLocaleString() : "—"}</td>
                      <td>{acc.exhaustedUntil ? new Date(acc.exhaustedUntil).toLocaleString() : "—"}</td>
                      <td>
                        <Button
                          size="sm"
                          variant="outline"
                          onClick={() => toggleMutation.mutate({ accountId: acc.id, status: acc.status === "disabled" ? "live" : "disabled" })}
                          disabled={toggleMutation.isPending}
                        >
                          {acc.status === "disabled" ? "Enable" : "Disable"}
                        </Button>
                      </td>
                    </tr>
                  ))}
                </tbody>
              </table>
            )}
          </section>

          <section>
            <h2 className="text-lg font-medium mb-2">Rotation history</h2>
            {historyQuery.isLoading ? (
              <p>Loading...</p>
            ) : historyQuery.data?.entries.length === 0 ? (
              <p className="text-gray-500">No rotations yet.</p>
            ) : (
              <ul className="space-y-2">
                {historyQuery.data?.entries.map((entry) => (
                  <li key={entry.id} className="border-l-2 pl-2 text-sm">
                    <span className="text-gray-400">{new Date(entry.createdAt).toLocaleString()}</span>
                    {" — "}
                    <span>Agent {entry.details.agentId.slice(0, 8)}</span>
                    {" rotated "}
    ```
