Integrar o serviço de contas (05-04) ao loop principal do heartbeat: antes de cada spawn de adapter `claude_local`, selecionar conta ativa e injetar `claudeConfigDir` no runtime config (MULTI-07).    
    ```typescript
    import type { Db } from "@paperclipai/db";
    import {
      claudeAccountsService,
      type ClaudeAccount,
      type ClaudeQuotaSubType,
    } from "./claude-accounts.js";
    import { NoAccountsAvailableError } from "./errors.js";
    import { detectClaudeQuotaExhausted } from "@paperclipai/adapter-claude-local/server";

    /**
     * MULTI-08 (Phase 5): Claude account swap orchestration.
     * See: server/src/services/claude-accounts-swap.md for design rationale.
     *
     * Flow:
     *   1. Classify exhaustion via detectClaudeQuotaExhausted
     *   2. Rotate via claudeAccountsService.rotateOnQuotaExhausted (DB updates only)
     *      → returns { rotationId, newAccount } — NO activity log emit yet (W1)
     *   3. Capture continuation summary (issue-continuation-summary primitive)
     *   4. Try Strategy A: resume with --resume <session_id> on new account
     *   5. Detect resume failure (D-22), fall back to Strategy B (full-context re-prompt)
     *   6. Return result for caller (heartbeat) to invoke retry with new account + prompt
     *
     * **W1 (revision r2):** activity log emission is DEFERRED. orchestrateClaudeSwap
     * returns rotationId; caller (heartbeat) MUST call
     * claudeAccountsService.recordSwapOutcome({ rotationId, swapStrategy, swapStatus })
     * AFTER Strategy A or B settles. This ensures D-32 observability matches reality.
     */

    export type SwapReason =
      | "swapped"
      | "no_detection"
      | "pool_empty"
      | "rotation_cap_reached"
      | "missing_credentials";

    export type SwapStrategy = "resume" | "fallback_full_context";

    export interface OrchestrateClaudeSwapInput {
      db: Db;
      agentId: string;
      companyId: string;
      runId: string;
      stepId: string;
      fromAccountId: string;
      previousSessionId: string | null; // session_id do step que falhou
      adapterStdout: string;
      adapterStderr: string;
      continuationSummary: string | null;     // null se issue não tem summary ainda
      alreadyRotatedThisStep: boolean;        // D-19 guard
      actorId: string;
    }

    export interface OrchestrateClaudeSwapResult {
      swapped: boolean;
      reason: SwapReason;
      rotationId: string | null;              // W1: handle to finalize via recordSwapOutcome
      newAccount: ClaudeAccount | null;
      newCredentialDir: string | null;
      strategy: SwapStrategy | null;          // intended strategy; caller may override to fallback_full_context
      retryPrompt: string | null;             // null para Strategy A; string para Strategy B
      retrySessionId: string | null;          // previousSessionId para Strategy A; null para Strategy B
      detectedType: ClaudeQuotaSubType | null;
    }

    export async function orchestrateClaudeSwap(
      input: OrchestrateClaudeSwapInput,
    ): Promise<OrchestrateClaudeSwapResult> {
      // D-19: Guard — no more than 1 rotation per step.
      if (input.alreadyRotatedThisStep) {
        return emptyResult("rotation_cap_reached");
      }

      // Step 1: Classify
      const detection = detectClaudeQuotaExhausted({
        stdout: input.adapterStdout,
        stderr: input.adapterStderr,
      });
      if (!detection.detected || detection.type === null) {
        return emptyResult("no_detection");
      }

      const svc = claudeAccountsService(input.db);

      // Step 2: Rotate (DB updates only). W1: do NOT pass swapStrategy here;
      // recordSwapOutcome is called by heartbeat after retry resolves.
      let rotation: { rotationId: string; newAccount: ClaudeAccount };
      try {
        rotation = await svc.rotateOnQuotaExhausted({
          agentId: input.agentId,
          companyId: input.companyId,
          fromAccountId: input.fromAccountId,
          errorFamily: detection.type,
          retryNotBefore: detection.retryAt,
          actorId: input.actorId,
          // NOTE (W1): swapStrategy is NOT passed. rotateOnQuotaExhausted
          // performs DB updates and creates a pending rotation record.
          // Heartbeat calls recordSwapOutcome(rotationId, effectiveStrategy, status)
          // AFTER Strategy A or B settles. D-32 accuracy.
        });
      } catch (err) {
        if (err instanceof NoAccountsAvailableError) {
          return emptyResult("pool_empty");
        }
        throw err;
      }

      // Step 3: Resolve credential dir for new account
      let newCredentialDir: string;
      try {
        newCredentialDir = await svc.resolveCredentialDir(rotation.newAccount);
      } catch (err) {
        // Conta foi marcada ativa mas operador não criou o dir → operacional, não código.
        // Caller will call recordSwapOutcome with status='failed' to finalize observability.
        return {
          swapped: true, // rotation persisted, but spawn impossível agora
          reason: "missing_credentials",
          rotationId: rotation.rotationId,
          newAccount: rotation.newAccount,
          newCredentialDir: null,
          strategy: null,
          retryPrompt: null,
          retrySessionId: null,
          detectedType: detection.type,
        };
      }

      // Step 4 & 5: Build retry plan — Strategy A first (resume).
      // Caller (heartbeat) executes the retry; if it detects resume failure via
      // detectResumeFailed(), it calls orchestrateFallbackFullContext for Strategy B.
      // Either way, caller calls recordSwapOutcome(rotationId, effectiveStrategy, status)
      // AFTER retry settles.
      return {
        swapped: true,
        reason: "swapped",
        rotationId: rotation.rotationId,
        newAccount: rotation.newAccount,
        newCredentialDir,
        strategy: "resume",
        retryPrompt: null,
        retrySessionId: input.previousSessionId,
        detectedType: detection.type,
      };
    }

    /**
     * Called by heartbeat when Strategy A failed (detected post-spawn).
     * Constructs Strategy B prompt. Does NOT emit activity log — caller does that
     * via recordSwapOutcome(rotationId, 'fallback_full_context', status).
     */
    export async function orchestrateFallbackFullContext(input: {
      db: Db;
      agentId: string;
      companyId: string;
      newAccount: ClaudeAccount;
      newCredentialDir: string;
      continuationSummary: string | null;
      lastInstruction: string | null;
      actorId: string;
    }): Promise<{ retryPrompt: string; strategy: SwapStrategy }> {
      const summary = input.continuationSummary?.trim() ?? "(no continuation summary captured before swap)";
      const lastInstr = input.lastInstruction?.trim() ?? "(no recorded last instruction; agent should re-read the issue)";

      const retryPrompt = [
        "Continuation of prior work (account swap due to Claude quota exhaustion).",
        "",
        "Previous progress summary:",
        summary,
        "",
        "Resume from:",
        lastInstr,
      ].join("\n");

      // W1: This function does NOT emit activity log. Heartbeat calls
      // claudeAccountsService.recordSwapOutcome(rotationId, 'fallback_full_context', status)
      // AFTER the Strategy B retry resolves.
      return { retryPrompt, strategy: "fallback_full_context" };
    }

    /**
     * Pattern detector for Strategy A failure (D-22).
     * Returns true if stdout/stderr indicates the resume failed.
     */
    export function detectResumeFailed(input: {
      stdout: string;
      stderr: string;
      exitCode: number | null;
    }): boolean {
      const blob = `${input.stdout}\n${input.stderr}`;
      if (/session\s+not\s+found|invalid\s+session\s+id|unknown\s+session/i.test(blob)) {
        return true;
      }
      // Exit code != 0 BEFORE any "system":"init" event in stdout JSON stream
      if (input.exitCode !== null && input.exitCode !== 0) {
        const hasInit = /"type":\s*"system"[\s\S]*?"subtype":\s*"init"/.test(input.stdout);
        if (!hasInit) return true;
      }
      return false;
    }

    function emptyResult(reason: SwapReason): OrchestrateClaudeSwapResult {
      return {
        swapped: false,
        reason,
        rotationId: null,
        newAccount: null,
        newCredentialDir: null,
        strategy: null,
        retryPrompt: null,
        retrySessionId: null,
        detectedType: null,
      };
    }
    ```
