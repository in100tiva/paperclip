```typescript
// ATUAL (a SUBSTITUIR):
function mapInviteAuthFeedback(error: unknown, mode: "sign-up" | "sign-in", email: string): string {
  if (error instanceof BetterAuthError) {
    if (error.code === "USER_ALREADY_EXISTS_USE_ANOTHER_EMAIL") {
      return `An account already exists for ${email}. Sign in below to continue.`;
    }
    if (error.code === "INVALID_EMAIL_OR_PASSWORD") {
      return "Email or password do not match an existing Paperclip account.";
    }
  }
  if (error instanceof Error && error.message.includes("Request failed: 401")) {
    return "Invalid email or password.";
  }
  // ... etc
}

// NOVO (target):
function mapInviteAuthFeedback(error: unknown, t: TFunction, email: string): string {
  if (error instanceof BetterAuthError) {
    const codeMap: Record<string, string> = {
      "USER_ALREADY_EXISTS_USE_ANOTHER_EMAIL": "errors:auth.user-already-exists",
      "INVALID_EMAIL_OR_PASSWORD": "errors:auth.invalid-email-or-password",
    };
    const key = codeMap[error.code as string];
    if (key) return t(key, { email });
  }
  if (error instanceof Error) {
    if (error.message.includes("Request failed: 401")) return t("errors:auth.request-failed-401");
    if (error.message.includes("Request failed: 422")) return t("errors:auth.request-failed-422", { email });
  }
  return t("errors:generic.unknown");
}
```
