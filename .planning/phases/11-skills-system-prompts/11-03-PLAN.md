async function materializeSkillForLocale(
  source: string,
  target: string,
  locale: RuntimeLocale
): Promise<void> {
  const variantPath = path.join(source, `SKILL.${locale}.md`);
  const hasVariant = await fs.access(variantPath).then(() => true).catch(() => false);
  if (!hasVariant || locale === "en-US") {
    return ensurePaperclipSkillSymlink(source, target);
  }
  await copyDirRecursive(source, target);
  await fs.rename(path.join(target, `SKILL.${locale}.md`), path.join(target, "SKILL.md"));
}

       export async function materializeSkillForLocale(
         source: string,
         target: string,
         locale: "pt-BR" | "en-US"
       ): Promise<void> {
         const variantPath = path.join(source, `SKILL.${locale}.md`);
         const hasVariant = await fs.access(variantPath).then(() => true).catch(() => false);
         if (!hasVariant || locale === "en-US") {
           return ensurePaperclipSkillSymlink(source, target);
         }
         await copyDirRecursive(source, target);
         await fs.rename(
           path.join(target, `SKILL.${locale}.md`),
           path.join(target, "SKILL.md")
         );
       }

       async function copyDirRecursive(source: string, target: string): Promise<void> {
         await fs.mkdir(target, { recursive: true });
         const entries = await fs.readdir(source, { withFileTypes: true });
         for (const entry of entries) {
           const srcPath = path.join(source, entry.name);
           const dstPath = path.join(target, entry.name);
           if (entry.isDirectory()) {
             await copyDirRecursive(srcPath, dstPath);
           } else if (entry.isFile()) {
             await fs.copyFile(srcPath, dstPath);
           }
         }
       }
