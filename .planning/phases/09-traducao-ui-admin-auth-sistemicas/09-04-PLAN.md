import { readFileSync } from "fs";
import { sync as glob } from "glob";

describe("confirm strings lint", () => {
  it("all window.confirm() calls use t()", () => {
    const files = glob("ui/src/**/*.{ts,tsx}", { ignore: ["**/__tests__/**", "**/*.test.*"] });
    const violations: string[] = [];
    for (const file of files) {
      const content = readFileSync(file, "utf-8");
      // Match window.confirm( followed by anything that's NOT t( or a translated string
      const regex = /window\.confirm\(\s*(?!t\(|t\s*\()/g;
      const matches = content.match(regex);
      if (matches) violations.push(`${file}: ${matches.length} raw window.confirm()`);
    }
    expect(violations).toEqual([]);
  });
});

       import { readFileSync, readdirSync, statSync } from "fs";
       import { join } from "path";
       import { describe, it, expect } from "vitest";

       function walkSrc(dir: string, files: string[] = []): string[] {
         for (const entry of readdirSync(dir)) {
           const full = join(dir, entry);
           const stat = statSync(full);
           if (stat.isDirectory()) {
             if (entry === "__tests__" || entry === "node_modules") continue;
             walkSrc(full, files);
           } else if (/\.(ts|tsx)$/.test(entry) && !/\.test\./.test(entry)) {
             files.push(full);
           }
         }
         return files;
       }

       describe("confirm-strings lint", () => {
         it("all window.confirm() calls wrap a t() call", () => {
           const files = walkSrc("ui/src");
           const violations: string[] = [];
           for (const file of files) {
             const content = readFileSync(file, "utf-8");
             // Match window.confirm( NOT followed by t( or t\s*(
             const regex = /window\.confirm\(\s*(?!t\(|t\s+\()/g;
             let match;
             while ((match = regex.exec(content)) !== null) {
               const lineNum = content.slice(0, match.index).split("\n").length;
               violations.push(`${file}:${lineNum}`);
             }
           }
           expect(violations, `Raw window.confirm() found:\n${violations.join("\n")}`).toEqual([]);
         });

         it("at least one window.confirm(t(...)) exists (sanity)", () => {
           const files = walkSrc("ui/src");
           let count = 0;
           for (const file of files) {
             const content = readFileSync(file, "utf-8");
             count += (content.match(/window\.confirm\(t\(/g) ?? []).length;
           }
           expect(count).toBeGreaterThan(0);
         });
       });
